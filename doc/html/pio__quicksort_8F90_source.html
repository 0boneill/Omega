00055 RECURSIVE SUBROUTINE quick_sort_1(left_end, right_end)
00056 
00057 implicit none
00058 INTEGER, INTENT(IN) :: left_end, right_end
00059 
00060 !     Local variables
00061 INTEGER             :: i, j, itemp
00062 integer                :: reference, temp
00063 INTEGER, PARAMETER  :: max_simple_sort_size = 6
00064 
00065 IF (right_end < left_end + max_simple_sort_size) THEN
00066   ! Use interchange sort for small lists
00067   CALL interchange_sort(left_end, right_end)
00068 
00069 ELSE
00070   ! Use partition ("quick") sort
00071   reference = list((left_end + right_end)/2)
00072   i = left_end - 1; j = right_end + 1
00073 
00074   DO
00075     ! Scan list from left end until element >= reference is found
00076     DO
00077       i = i + 1
00078       IF (list(i) >= reference) EXIT
00079     END DO
00080     ! Scan list from right end until element <= reference is found
00081     DO
00082       j = j - 1
00083       IF (list(j) <= reference) EXIT
00084     END DO
00085 
00086 
00087     IF (i < j) THEN
00088       ! Swap two out-of-order elements
00089       temp = list(i); list(i) = list(j); list(j) = temp
00090       itemp = order(i); order(i) = order(j); order(j) = itemp
00091     ELSE IF (i == j) THEN
00092       i = i + 1
00093       EXIT
00094     ELSE
00095       EXIT
00096     END IF
00097   END DO
00098 
00099   IF (left_end < j) CALL quick_sort_1(left_end, j)
00100   IF (i < right_end) CALL quick_sort_1(i, right_end)
00101 END IF
00102 
00103 END SUBROUTINE quick_sort_1
